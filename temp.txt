public enum TargetType {
    AprilTags,
    Notes
};


TargetType targetType = Notes;


// pick the spatially lowest apriltag or note as selected by the user
int lowestI;
double lowestTync = 100;  // larger than any possible tync
if (targetType == AprilTags) {
    // track apriltags
    RawFiducial[] fiducials = getRawFiducials("limelight");
    // find the fiducial with highest tync value (which is spatially lowest)
    for (int i = 0; i < fiducials.length; i++) {
        if (fiducials[i].tync < lowestTync) {
            lowestTync = fiducials[i].tync;
            lowestI = i;
        }
    }
    // record the TX, TY, and TA from the fiducial with highest tync value
    TX = fiducials[lowestI].txnc;
    TY = fiducials[lowestI].tync;
    TA = fiducials[lowestI].ta;
} else {
    // track notes
    RawDetection[] detections = getRawDetections("limelight");
    // find the detection with highest tync value (which is spatially lowest)
    for (int i = 0; i < detections.length; i++) {
        if (detections [i].tync < lowestTync) {
            lowestTync = detections [i].tync;
            lowestI = i;
        }
    }
    // record the TX, TY, and TA from the detection with highest tync value
    TX = detections [lowestI].txnc;
    TY = detections [lowestI].tync;
    TA = detections [lowestI].ta;
}
